'isCarNumber': {
  "func": function(field,rules,i,options){
      var status = true
      var vehicleNumber = field.val();
      var xreg=/^[京津沪渝冀豫辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤青藏川宁琼使领A-Z]{1}[A-Z]{1}(([0-9]{5}[DF]$)|([DF][A-HJ-NP-Z0-9][0-9]{4}$))/;// 新能源车规则

      var licensePlate = vehicleNumber.replace(/，/ig,',');  // 如有输入中文"，" 则替换成英文','
      var licensePlateList = licensePlate.split(',');
      for(var i = 0 ;i<licensePlateList.length; i++){
        if(licensePlateList[i]){
          if(licensePlateList[i].length == 7){
            status =  creg.test(licensePlateList[i]);
          } else if(licensePlateList[i].length == 8){
            status = xreg.test(licensePlateList[i]);
          } else{
            status = false
          }
        }
      }
      return status;
  },
  "alertText":"* 存在错误的车牌号码"
